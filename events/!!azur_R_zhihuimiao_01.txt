country_event = {
	id = azur_mao.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		has_country_flag = azur_lane_gangqu_country
		NOR = { 
			has_country_flag = azur_zhihuimao_come
			exists = event_target:azur_lane_mao_speics
		}
	}
	
	immediate = {
		set_country_flag = azur_zhihuimao_come
		create_species = {
			name = 指挥猫
			plural = 指挥猫
			class = MAM
			portrait = 奥古喵
			namelist = AZURLANE
			immortal = yes
			traits = {
				#特质
				trait = trait_talented
			}
			effect = {
				save_global_event_target_as = azur_lane_mao_speics
			}
		}
	}
}
