#母星重建
planet_event = {
	id = azur_R_origin.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		owner = {
			has_country_flag = azur_lane_gangqu_country
		}
		is_planet_class = pc_azur_city_R
		has_planet_flag = azur_world
	}
	
	immediate = {
		while = {
			count = 19
			create_pop = {
				species = owner_main_species
			}
		}
		set_capital = yes
		owner = {
			country_event = { id = azur_R_origin.3	days = 1 }
		}
	}
}
