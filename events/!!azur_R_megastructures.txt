country_event = {
	id = "azur_R_megastructures.4"
	title = "azur_d.4.name"
	desc = {
		text = azur_d.4.desc.a
		trigger = {
			NOT = { has_country_flag = azur_r_built_ringworld }
		}
	}
	desc = {
		text = azur_d.4.desc.b
		trigger = {
			has_country_flag = azur_r_built_ringworld
		}
	}
	show_sound = event_ringworld_build_complete
	picture = GFX_evt_ringworld
	is_triggered_only = yes
	location = FROM
    
    after = {
        if = {
            limit = { NOT = { has_country_flag = azur_r_built_ringworld } }
            set_country_flag = azur_r_built_ringworld
        }
	}

	option = {
		name = azur_d.4.a
	}
	
	option = {
		name = azur_d.4.b
		trigger = {
			NOT = {
				has_country_flag = azur_r_built_ringworld
			}
		}
		hidden_effect = {
			custom_tooltip = azur_d.4.b.tooltip
		}
	}
	
	option = {
		name = azur_d.4.c
	}
}
