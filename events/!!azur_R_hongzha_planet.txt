namespace = azur_lane_hongzha

planet_event = {
	id = azur_lane_hongzha.1
	hide_window = yes
    is_triggered_only = yes
    trigger = {
		NOT = { has_planet_flag = azur_lane_planet_mubiao }
		NOT = { has_planet_flag = azur_fallen_empire_world }
		From = { 
			OR = {
				has_authority = auth_azur_dictatorial
			}
			has_civic = civic_azur_R_03
			any_owned_fleet = {
				is_ship_class = shipclass_military
				orbit = { is_same_value = ROOT }
				fleet_size >= 120
			}
		}
    }
	immediate = {
		save_event_target_as = azur_planet_hongzha
		set_timed_planet_flag = { flag = azur_lane_planet_mubiao months = 2 }
		From = { 					
			random_owned_fleet = {
				limit = {
					is_ship_class = shipclass_military
					orbit = { is_same_value = event_target:azur_planet_hongzha }				
				}	
				save_event_target_as = azur_lane_hongzha_fleet
			}
			country_event = { id = azur_lane_hongzha.2 days = 3 }	
		}
	}	
}

planet_event = {
	id = azur_lane_hongzha.5
	hide_window = yes
    is_triggered_only = yes
	
    trigger = {
		has_planet_flag = azur_fallen_empire_world
		From = { 
			any_owned_fleet = {
				is_ship_class = shipclass_military
				orbit = { is_same_value = ROOT }
				fleet_size >= 1
			}
		}
    }
	
	immediate = {
		save_event_target_as = azur_fallen_planet
		From = { 					
			random_owned_fleet = {
				limit = {
					is_ship_class = shipclass_military
					orbit = { is_same_value = event_target:azur_fallen_planet }				
				}	
				save_event_target_as = azur_lane_ship_remove
			}
			country_event = { id = azur_lane_hongzha.6 days = 1 }
			country_event = { id = azur_lane_hongzha.7 days = 1 }			
		}
	}	
}
