country_event = {
	id = azur_leader_lysm.1
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		has_country_flag = azur_lane_gangqu_country
	}
	
	immediate = {
		#获得领袖
		clone_leader = {
			target = event_target:萊妮絲_leader
		}
		last_created_leader = {
			change_leader_portrait = AL_R_leader_萊妮絲_01
			set_leader_flag = azur_leader_萊妮絲  					#领袖标记
			save_global_event_target_as = 萊妮絲_leader_country
			set_owner = root
			set_leader_flag = legendary_leader
			set_leader_flag = immune_to_negative_traits
			set_skill = 20
			add_trait_no_notify = random_common
			remove_all_traits = yes
			add_trait_no_notify = leader_azur_trait_Reines_01
			add_trait_no_notify = leader_azur_trait_Reines_02
			add_trait_no_notify = leader_azur_trait_Reines_03
			add_trait_no_notify = leader_azur_trait_Reines_04
			add_trait_no_notify = leader_azur_trait_Reines_05
			add_trait_no_notify = leader_azur_trait_Reines_06
			add_trait_no_notify = leader_azur_trait_Reines_07
			add_trait_no_notify = leader_azur_trait_Reines_08
			set_leader_flag = should_not_have_upkeep
		}
		
		#变量
		#资源相关
		set_variable = { which = azur_R_Reines_resource value = 0 }
		set_variable = { which = azur_R_Reines_resource_give value = 0 }
		set_variable = { which = azur_R_Reines_resource_income value = 0 }
		#技能开关
		set_country_flag = azur_R_Reines_battle_ON
	}
}

	hide_window = yes
	
	trigger = {
		has_country_flag = azur_lane_gangqu_country
		has_country_flag = azur_R_XI_has
		exists = event_target:希_leader
	}
	
	immediate = {
		if = {
			limit = {
				event_target:希_leader = {
					OR = { 
						is_researching_area = physics
						is_researching_area = society
						is_researching_area = engineering
					}
				}
			}
			change_variable = {
				which = azur_R_XI_love
				value = -1
			}
		}
	}
}

		name = azur_leader_lysm.117.a

			custom_tooltip = {

				is_at_war_with = event_target:azur_R_fallen_empire
			}
		}
		trigger = {
			NOT = {
				has_country_flag = azur_R_fallen_level_down
			}

		is_dialog_only = yes
		response_text = "azur_leader_lysm.117.a.desc"

		allow = {
			custom_tooltip = {
				fail_text = azur_leader_lysm.117.b.allow
				end_game_years_passed >= 0
			}
			custom_tooltip = {
				fail_text = azur_leader_lysm.104.love_200
				check_variable = {
					which = azur_R_XI_love
					value >= 200
				}
			}
			resource_stockpile_compare = { resource = minor_artifacts value >= 90000 }
		}
		trigger = {
			NOT = {
				has_country_flag = azur_R_fallen_tech
			}
		}
		is_dialog_only = yes
		response_text = "azur_leader_lysm.117.b.desc"
		add_resource = { minor_artifacts = -90000 }
		hidden_effect = {
			custom_tooltip = azur_leader_lysm.117.b.tooltip
			add_research_option = tech_azur_R_fallen_X
			add_research_option = tech_azur_R_fallen_L
			add_research_option = tech_azur_R_fallen_M
			add_research_option = tech_azur_R_fallen_S
			add_research_option = tech_azur_R_fallen_UTILITY
			add_research_option = tech_azur_R_fallen_JUMP_DRIVE
			set_country_flag = azur_R_fallen_tech
		}
	}
	
	#其他
	
	option = { #关闭界面
		name = azur_leader_lysm.104.close
		hidden_effect = {
			country_event = { id = azur_leader_lysm.112 }
		}
		default_hide_option = yes
	}
}

#战斗胜利加层数
country_event = {
	id = azur_leader_lysm.118
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		This = {
			has_country_flag = azur_lane_gangqu_country
			has_country_flag = azur_R_XI_relic
		}
	}
	
	immediate = {
		#增加层数
		change_variable = {
			which = azur_R_XI_relic
			value = 4
		}
		remove_modifier = AL_R_XI_RELICS_01
		add_modifier = {
			modifier = "AL_R_XI_RELICS_01"
			multiplier = azur_R_XI_relic
		}
	}
}
