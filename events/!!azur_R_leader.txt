namespace = azur_leader
#开局
country_event = {
	id = azur_leader.1
	is_triggered_only = yes
	diplomatic = yes
	custom_gui = "azur_lane_r_newUI_1"
	
	trigger = {
		owner_main_species = {
			has_trait = trait_azur_a1
		}
	}
	
	immediate = {
		#国家标记
		set_country_flag = azur_lane_gangqu_country
		#全局检测
		set_global_flag = 碧蓝航线_R
		
		#领袖选择界面
		set_country_flag = azur_lane_R_leader_界面_1
		
		#移除开局领袖
		random_owned_leader = {
			limit = { 
				is_councilor = yes
				is_ruler = yes
			}
			save_global_event_target_as = azur_leader_ruler
		}
		#种族
		create_species = {
			name = 虚拟歌姬
			plural = 虚拟歌姬
			class = MACHINE
			portrait = AL_leader_初音未来
			namelist = AZURLANE
			immortal = yes
			traits = {
				#特质
				trait = trait_robot_emotion_emulators
			}
			effect = {
				save_global_event_target_as = azur_lane_vocaloid
			}
		}
		#领袖种族
		create_species = {
			name = "舰娘"
			class = "Azur_Lane_R"
			portrait = "Azur_Lane_R"
			traits = {
				trait = trait_azur_a1
			}
			namelist = AZURLANE
			immortal = yes
			effect = {
				save_global_event_target_as = azur_lane_leader_species
			}
		}
		#摸鱼飞弹
		give_technology = { tech = tech_azur_my_fd message = no }
		#彩蛋
		set_variable = { which = azur_R_nanoda value = 0 }
		#希变量相关
		#图书馆等级
		set_variable = { which = azur_R_XI_level value = 1 }
		#资源上限补正
		add_modifier = {
			modifier = AL_R_resources_max
			days = -1
		}
	}
	
	after = {
		#(雪风，绫波，恶毒）
		country_event = { id = azur_leader.10 }
		country_event = { id = azur_leader.11 }
		country_event = { id = azur_leader.12 }
		if = { 
			limit = { has_country_flag = AL_R_leader_set_长门 }
			hidden_effect = {
				country_event = { id = azur_leader.13 }
				set_country_flag = azur_lane_leader_changmen
				#局势、御神木
				country_event = { id = azur_R_nagato.2 }
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_克莱蒙梭 }
			hidden_effect = {
				country_event = { id = azur_leader.14 }
				set_country_flag = azur_lane_leader_clemenceau
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_腓特烈大帝 }
			hidden_effect = {
				country_event = { id = azur_leader.15 }
				set_country_flag = azur_lane_leader_feitelie
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_君主 }
			hidden_effect = {
				country_event = { id = azur_leader.16 }
				set_country_flag = azur_lane_leader_junzhu
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_维内托 }
			hidden_effect = {
				country_event = { id = azur_leader.17 }
				set_country_flag = azur_lane_leader_weineituo
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_黎塞留 }
			hidden_effect = {
				country_event = { id = azur_leader.18 }
				set_country_flag = azur_lane_leader_lisailiu
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_企业 }
			hidden_effect = {
				country_event = { id = azur_leader.19 }
				set_country_flag = azur_lane_leader_qiye
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_苏维埃同盟 }
			hidden_effect = {
				country_event = { id = azur_leader.20 }
				set_country_flag = azur_lane_leader_suweiaitongmeng
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_逸仙 }
			hidden_effect = {
				country_event = { id = azur_leader.21 }
				set_country_flag = azur_lane_leader_yixian
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_伊丽莎白 }
			hidden_effect = {
				country_event = { id = azur_leader.22 }
				set_country_flag = azur_lane_leader_elizabeth
			}
		}
		if = { 
			limit = { has_country_flag = AL_R_leader_set_零 }
			hidden_effect = {
				country_event = { id = azur_siren_zero.2 }
				country_event = { id = azur_siren_zero.1 }
				set_country_flag = azur_lane_leader_zero
			}
		}
		
		
		#绿毛
		if = { 
			limit = { has_country_flag = AL_R_leader_set_初音未来 }
			hidden_effect = {
				country_event = { id = azur_R_vocaloid.01 }		#MIKU
				set_country_flag = azur_lane_leader_miku
				country_event = { id = azur_R_vocaloid.02 }		#LUKA
			}
		}
		#开局领袖
		if = {
			limit = { has_country_flag = AL_R_leader_has_set }
			event_target:azur_leader_ruler = {
				exile_leader_as = azur_leader_ruler
			}
		}
		#隐藏菜单
		if = {
			limit = {
				check_variable = {
					which = azur_R_nanoda
					value = 10
				}
				NOT = { has_global_flag = azur_R_开启隐藏菜单_全局 }
			}
			set_global_flag = azur_R_开启隐藏菜单_全局
			set_country_flag = azur_R_开启隐藏菜单
		}
	}
	
	option = { #
		name = azur_event_opinion
		default_hide_option = yes
	}
}

#雪风
country_event = {
	id = azur_leader.10
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		has_country_flag = azur_lane_gangqu_country
	}
	
	immediate = {
		#创建雪风
		create_leader = {
			name = 雪风
			species = event_target:azur_lane_leader_species
			class = official  				#official/scientist/random_ruler/commander
			skill = 1
			set_age = 12
			traits = {
				trait = leader_azur_trait_xuefeng
			}

			set_is_female = yes
			event_leader = no
			hide_age = no

			custom_description = AL_R_xuefeng_backstory
			custom_catch_phrase = AL_R_xuefeng_phrase
			background_job = azur_R_leader_job_A
			background_ethic = ethic_spiritualist
		}
		last_created_leader = {
			change_leader_portrait = AL_leader_雪风
			set_leader_flag = azur_leader_xuefeng  #领袖标记
			save_global_event_target_as = xuefeng_leader #保存事件
			#新flag
			set_leader_flag = legendary_leader
			set_leader_flag = immune_to_negative_traits
			set_skill = 20
			while = {
				count = 9
				add_trait_no_notify = random_common
			}
			while = {
				count = 2
				add_leader_destiny_traits_AL_R = yes
			}
			set_leader_flag = should_not_have_upkeep
		}
	}
}
