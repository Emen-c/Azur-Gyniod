mega_azur_miku_0 = {
	entity = "construction_platform_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = yes
	entity_offset = { x = -12 y = -12 }
	build_time = 7200
	resources = {
		category = megastructures
		cost = {
			alloys = 6480
			sr_hongjianjian = 6480
			influence = 300
		}

		upkeep = {
			energy = 5
		}
	}

	prerequisites = { "tech_azur_miku_1" }

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			has_no_non_gate_megastructure = yes
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system"
			NOT = {
				any_system_planet = {
					is_surveyed = {
						who = prev.from
						status = no
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_mega_art_installation"
			from = { NOT = { has_country_flag = built_mega_azur_0 } }
		}
        custom_tooltip = {
			fail_text = "requires_no_existing_megastructure"
			NOR = { has_star_flag = ring_world_built }
		}
	}
	
	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {			# prevent leaking habitability information
					who = prev.from
					status = yes
				}
				colonizable_planet = no
			}
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "requires_not_minor_planetary_body"
				NOR = {
					is_asteroid = yes
					is_moon = yes
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
			custom_tooltip = {
				fail_text = "requires_not_ring_world"
				is_ringworld = no
			}
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				NOR = { 
					is_planet_class = pc_gas_giant
					has_planet_flag = megastructure
					has_planet_flag = has_megastructure
				}
			}
		}
	}
	
	# root = system
	# from = country	
	ai_weight = {
		factor = 0
		
	}
	
	on_build_start = {}
	on_build_cancel = {}
	on_build_complete = {
        set_star_flag = azur_r_miku_built
		from = { set_country_flag = built_mega_azur_0 }
		fromfrom.planet = {
			if = {
				limit = { has_orbital_station = yes }
				orbital_station = {
					dismantle = yes
				}
			}
			set_planet_flag = has_megastructure
		}
	}
}

	}

	}

	show_prereqs = no

		planet_amenities_mult = 0.39
	}

	on_build_complete = {
		set_star_flag = mega_azur_miku_installation_built
		from = {
			set_country_flag = has_built_or_repaired_megastructure

		}

			if = {
				limit = {
					NOT = {
						any_system_ambient_object = { has_ambient_object_flag = extradimensional_system_effect }
					}
				}
				fromfrom.planet = {
					create_ambient_object = {
						type = "azur_R_miku_mega_01"
						location = this
					}
					last_created_ambient_object = {
						set_ambient_object_flag = extradimensional_system_effect
						set_location = {
							target = prev
							distance = 0
							angle = 0
						}
					}
				}
			}
		}
	}
