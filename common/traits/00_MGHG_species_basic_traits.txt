# Scopes are this = species, from = country performing the modification.

#尼德霍格的馈赠
trait_you_are_monster_girl = {
	cost = 0
	sorting_priority = 19

	initial = yes
	randomized = no

	species_potential_add = {
		is_same_species = owner_main_species
		from = {
			is_ai = no
			NOT = {
				any_owned_species = {
					is_monster_girl = yes
				}
			}
		}
	}
	species_possible_remove = {
		from = {
			is_ai = no
			NOT = {
				any_owned_species = {
					is_monster_girl = yes
				}
			}
		}
	}
	species_possible_merge_add = {
		is_same_species = owner_main_species
		from = {
			is_ai = no
			NOT = {
				any_owned_species = {
					is_monster_girl = yes
				}
			}
		}
	}

	potential_crossbreeding_chance = 0

	opposites = {
		"trait_dickman"
		"trait_lamia"
		"trait_angel"
		"trait_Gynoid"
		"trait_slime"
		"trait_succubus"
		"trait_vampire"
		"trait_zombie_girl"
		"trait_angel_fallen"
		"trait_dragon"
	}

	allowed_archetypes = {
		BIOLOGICAL
		LITHOID
		MACHINE
		ELF
		Magical_Bitch
		Magical_Girl
		Lust_Girl
		TENTACLE
		LOVEMACHINE
		NYTO
		EILD
		WARSHIPGIRL
		SHENHAI
		UNITED_FLEET
		KANSEN
	}

	modifier = {
		pop_amenities_usage_mult = -0.1
		planet_pops_lust_crystal_upkeep_add = 0.1
	}

	custom_tooltip_with_modifiers = succubus_ap_tooltip
	triggered_planet_growth_habitability_modifier = {
		potential = {
			exists = owner
			owner = {
				has_ascension_perk = ap_live_long_and_prosper
			}
			exists = growing_species
		}
		pop_growth_speed = 0.25
	}

	randomized = no

	ai_weight = {
		weight = 0
	}

}
