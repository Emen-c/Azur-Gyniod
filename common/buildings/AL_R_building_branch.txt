#女王雕像
holding_AL_R_elizabeth_1 = {
	base_buildtime = 720
	owner_type = subject_holding
	icon = holding_AL_R_elizabeth_1
	base_cap_amount = 1

	category = government

	potential = {
		owner.overlord = {
			AND = {
				exists = event_target:elizabeth_leader
				has_country_flag = azur_lane_leader_elizabeth
			}
		}
	}
	
	allow = {
		custom_tooltip = {
			fail_text = subject_is_AL_R_elizabeth
			exists = planet.owner
			planet.owner = {
				is_specialist_subject_type = { TYPE = ALelizabeth }
			}
		}
	}
	
	#宗主国修正
	country_modifier = {
		diplo_weight_mult = 0.2
	}
	
	planet_modifier = {
		planet_stability_add = 5
	}

	triggered_planet_modifier = {
		potential = {
			exists = planet.owner
			planet.owner = { is_gestalt = no }
		}
		job_overlord_propagandist_add = 4
	}

	triggered_planet_modifier = {
		potential = {
			exists = planet.owner
			planet.owner = { is_gestalt = yes }
		}
		job_overlord_propagandist_drone_add = 4
	}

	resources = {
		category = planet_holdings
		cost = {
			minerals = 2000
		}
		upkeep = {
			energy = 10
		}
	}
	
	destroy_trigger = {
		exists = planet.owner
		planet.owner = {
			NOT = {
				is_specialist_subject_type = { TYPE = ALelizabeth }
			}
		}
	}

	triggered_desc = {
		text = holding_AL_R_elizabeth_1_job_desc
	}
}
