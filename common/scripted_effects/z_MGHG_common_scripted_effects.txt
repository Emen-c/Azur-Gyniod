create_synth_species_effect = {
	if = {
		limit = {
			NOT = {
				exists = event_target:synthspecies@root.owner
			}
		}
		if = {
			limit = {
				owner = {
					has_policy_flag = simple_Gynoid
				}
			}
			owner_main_species = {
				species = {
					modify_species = {
						effect = {
							create_species = {
								name = this
								namelist = this
								plural = this
								class = "GYNOID_ROBOT"
								portrait = "GYNOID02"
								traits = {
									trait = "trait_mechanical"
									trait = "trait_Gynoid"
									trait = "trait_cyber_Gynoid"
								}
								can_be_modified = this
								homeworld = this
								gender = this
	
								effect = {
									set_synth_origin_traits_effect = yes
								}
							}
	
							last_created_species = {
								save_global_event_target_as = synthspecies@root.owner
							}
						}
					}
				}
			}
		}
		else_if = {
			limit = {
				owner = {
					has_policy_flag = complex_Gynoid
				}
			}
			owner_main_species = {
				species = {
					modify_species = {
						effect = {
							switch = {
								trigger = is_species_class
								LAMIA = {
									synth_species_complex_gynoid_create = {
										SPECIES = LAMIA
									}
								}
								ANGEL = {
									synth_species_complex_gynoid_create = {
										SPECIES = ANGEL
									}
								}
								SLIME = {
									synth_species_complex_gynoid_create = {
										SPECIES = SLIME
									}
								}
								SUCCUBUS = {
									synth_species_complex_gynoid_create = {
										SPECIES = SUCCUBUS
									}
								}
								VAMPIRE = {
									synth_species_complex_gynoid_create = {
										SPECIES = VAMPIRE
									}
								}
								ANGEL_F = {
									synth_species_complex_gynoid_create = {
										SPECIES = ANGEL_F
									}
								}
								DRAGON = {
									synth_species_complex_gynoid_create = {
										SPECIES = DRAGON
									}
								}
							}
							last_created_species = {
								save_global_event_target_as = synthspecies@root.owner
							}
						}
					}
				}
			}
		}
		else = {
			owner_main_species = {
				species = {
					modify_species = {
						effect = {
							create_species = {
								is_mod = yes
								name = this
								plural = this
								class = "ROBOT"
								portrait = "default_robot"
								traits = {
									trait = trait_mechanical
								}
								can_be_modified = this
								homeworld = this
								namelist = this
								gender = this
	
								effect = {
									set_synth_origin_traits_effect = yes
								}
							}
	
							last_created_species = {
								set_synth_or_robot_portrait_effect = yes
								save_global_event_target_as = synthspecies@root.owner
							}
						}
					}
				}
			}
		}
		owner = {
			if = {
				limit = {
					has_policy_flag = robots_outlawed
				}
				set_policy = {
					policy = robot_pop_policy
					option = robot_pops_allowed
					cooldown = yes
				}
			}
			if = {
				limit = {
					OR = {
						has_policy_flag = ai_servitude
						has_policy_flag = ai_outlawed
					}
				}
				set_policy = {
					policy = artificial_intelligence_policy
					option = ai_full_rights
					cooldown = yes
				}
			}
		}
	}
}
