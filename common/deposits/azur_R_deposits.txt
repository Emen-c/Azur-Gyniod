d_azur_decrepit_dwellings = {
	icon = d_city_ruins
	time = 11

	is_for_colonizeable = yes
	category = deposit_cat_blockers

	potential = { always = no }

	on_cleared = {
		if = {
			limit = {
				any_owned_pop = {
					is_gestalt = no
					is_robot_pop = no
				}
			}
			random_owned_pop = {
				limit = {
					is_gestalt = no
					is_robot_pop = no
				}
				root = {
					create_pop = {
						species = prev
					}
				}
			}
		}
		else = {
			random_owned_pop = {
				root = {
					create_pop = {
						species = prev
					}
				}
			}
		}
	}

	resources = {
		category = deposit_blockers
		cost = {
			energy = 1
		}
	}
	drop_weight = {
		weight = 0
	}
}
