	combat_source_anchor = root						#战斗源锚（在计算该船的作战位置时，将用作源的锚，这是锚船的实际位置。）
	
	collision_awareness = 0.75						#碰撞意识
	collision_radius = 1.0							#乘以船舶尺寸中指定的半径

	attack_move_pattern = stay_at_range				#攻击移动模式（有效值）
	passive_move_pattern = stay_at_range			#被动移动模式

	desc = "CARRIER_BEHAVIOR_DESC"					#描述
}
